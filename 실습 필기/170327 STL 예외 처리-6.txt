// dynamic_cast는 언제 실패하는가?

class Animal {
public:
	virtual void move() const = 0;
};

class Dog : public Animal {
public:
	virtual void move() const override final {
		cout << "개가 뛴다" << endl;
	};
};

class Bird : public Animal {
public:
	virtual void move() const override final {
		cout << "새가 난다" << endl;
	};
};

int main()
{
	Animal* ap[]{ new Dog, new Bird, new Dog }; // 다형성

	for (int i = 0; i < 3; ++i) {
		ap[i]->move();
	}
	save();
}
