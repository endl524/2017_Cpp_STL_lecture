// smart pointer
// unique_ptr - 자원을 독점하는 포인터
// shared_ptr - 자원을 공유하는 포인터

class Model {
public:
	Model() { cout << "ctor" << endl; }
	~Model() { cout << "dtor" << endl; }
};

class u_ptr {
	Model* ptr;

public:
	explicit u_ptr(Model* p) : ptr(p) {} // 포인터 세이브
	~u_ptr() { delete[] ptr; }
};

int main()
{
	u_ptr p(new Model[1000]); // 생성자의 인자로 준다.
	// 이 포인터 p는 기존 포인터와 크기가 같음을 보장한다. (4byte)

	save();
}
