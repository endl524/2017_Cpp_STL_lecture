// smart pointer
// unique_ptr - 자원을 독점하는 포인터
// shared_ptr - 자원을 공유하는 포인터

#include <iostream>
#include <memory>
#include "save.h"
using namespace std;

inline void printVal(int n) { cout << n << endl; }
inline void printVal(int* p) { cout << *p << endl; }



int main()
{
	int n{ 410 };
	int* p = &n;

	printVal(3); // 3을 출력
	printVal(n); // 410을 출력
	printVal(p); // 410을 출력

	save();
}
